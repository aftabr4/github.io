<?php
function getGCD($a, $b)
{
    if ($b == 0)
    {
        return $a;
    }
    else
    {
        return getGCD($b, $a % $b);
    }
}
function getLCM($a, $b)
{
    return ($a * $b) / getGCD($a, $b);
}
?>
