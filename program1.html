Program 1: Display Prime Numbers (PHP Code)

<?php
$number=2;
$c=0;
while($number < 50)
{
    $div_count=0;
    for ($i=1; $i<=$number; $i++)
    {
        if(($number % $i) == 0)
        {
            $div_count++; 
        }
    }
    if($div_count < 3)
    {
        echo $number . " , ";
        $c++;
    }
    $number = $number + 1;
}
echo "<br> Number of prime numbers = " . $c;
?>
